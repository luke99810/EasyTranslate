# 学术PDF翻译工具 - 技术架构设计文档

## 一、架构概述

### 1.1 架构原则

1. **前后端分离**：前端负责UI交互，后端负责PDF处理
2. **模块化设计**：各功能模块独立，便于维护和扩展
3. **异步处理**：翻译任务异步执行，避免阻塞
4. **本地优先**：尽可能在本地处理，保护用户隐私

### 1.2 技术栈选型

| 层级 | 技术选型 | 说明 |
|-----|---------|------|
| 前端 | React 18 + Vite | 现代React开发，快速构建 |
| 前端样式 | TailwindCSS | 原子化CSS，快速开发 |
| 前端状态 | Zustand | 轻量级状态管理 |
| 后端 | Python 3.10 + FastAPI | 高性能异步Web框架 |
| PDF处理 | PyMuPDF + pdfplumber | PDF解析和文本提取 |
| 翻译服务 | DeepL/Google/OpenAI API | 多引擎支持 |
| 部署 | Docker + Nginx | 容器化部署 |

#### 3.2.2 核心模块设计

**PDF处理器 (pdf_processor.py)**

```python
class PDFProcessor:
    """PDF处理核心类"""
    
    def __init__(self, file_path: str):
        self.file_path = file_path
        self.doc = fitz.open(file_path)
    
    def extract_content(self) -> List[PageContent]:
        """提取PDF内容"""
        pages = []
        for page_num in range(len(self.doc)):
            page = self.doc[page_num]
            content = self._extract_page_content(page)
            pages.append(content)
        return pages
    
    def _extract_page_content(self, page: fitz.Page) -> PageContent:
        """提取单页内容"""
        # 提取文本块
        text_blocks = self._extract_text_blocks(page)
        # 提取图片
        images = self._extract_images(page)
        # 识别公式
        formulas = self._detect_formulas(text_blocks)
        
        return PageContent(
            page_number=page.number,
            text_blocks=text_blocks,
            images=images,
            formulas=formulas
        )
    
    def _detect_formulas(self, text_blocks: List[TextBlock]) -> List[Formula]:
        """识别LaTeX公式"""
        formulas = []
        pattern = r'\$[^$]+\$|\$\$[^$]+\$\$|\\\([^)]+\\\)|\\\[[^\]]+\\\]'
        for block in text_blocks:
            matches = re.finditer(pattern, block.text)
            for match in matches:
                formulas.append(Formula(
                    content=match.group(),
                    position=match.span()
                ))
        return formulas
```

**翻译引擎 (translation_engine.py)**

```python
class TranslationEngine:
    """翻译引擎管理类"""
    
    def __init__(self, provider: TranslationProvider, api_key: str):
        self.provider = provider
        self.api_key = api_key
        self.service = self._get_service()
    
    def _get_service(self) -> TranslationService:
        """获取翻译服务实例"""
        services = {
            TranslationProvider.DEEPL: DeepLService,
            TranslationProvider.GOOGLE: GoogleService,
            TranslationProvider.OPENAI: OpenAIService,
        }
        return services[self.provider](self.api_key)
    
    async def translate(
        self, 
        text: str, 
        source_lang: str = 'en', 
        target_lang: str = 'zh'
    ) -> str:
        """翻译文本"""
        # 保护公式不被翻译
        protected_text, formulas = self._protect_formulas(text)
        # 翻译
        translated = await self.service.translate(
            protected_text, source_lang, target_lang
        )
        # 恢复公式
        result = self._restore_formulas(translated, formulas)
        return result
    
    def _protect_formulas(self, text: str) -> Tuple[str, List[str]]:
        """保护公式，用占位符替换"""
        formulas = []
        pattern = r'(\$[^$]+\$|\$\$[^$]+\$\$)'
        
        def replace_formula(match):
            formulas.append(match.group())
            return f"[[FORMULA_{len(formulas)-1}]]"
        
        protected = re.sub(pattern, replace_formula, text)
        return protected, formulas
```

**布局分析器 (layout_analyzer.py)**

```python
class LayoutAnalyzer:
    """PDF布局分析器"""
    
    def analyze(self, page: fitz.Page) -> Layout:
        """分析页面布局"""
        # 获取页面尺寸
        page_rect = page.rect
        
        # 提取文本块及其位置
        blocks = self._extract_blocks(page)
        
        # 分析列布局
        columns = self._detect_columns(blocks, page_rect.width)
        
        # 识别段落结构
        paragraphs = self._detect_paragraphs(blocks)
        
        return Layout(
            page_size=(page_rect.width, page_rect.height),
            columns=columns,
            paragraphs=paragraphs,
            blocks=blocks
        )
    
    def _detect_columns(self, blocks: List[Block], page_width: float) -> int:
        """检测列数（单栏/双栏）"""
        # 分析文本块的x坐标分布
        x_positions = [block.bbox[0] for block in blocks]
        # 使用聚类算法判断列数
        # 简化实现：检查是否有明显的左右分布
        left_blocks = sum(1 for x in x_positions if x < page_width / 2)
        right_blocks = sum(1 for x in x_positions if x >= page_width / 2)
        
        if left_blocks > 0 and right_blocks > 0:
            return 2
        return 1
```

### 3.3 数据模型

```python
# schemas.py

from pydantic import BaseModel
from typing import List, Optional, Tuple
from enum import Enum

class TranslationProvider(str, Enum):
    DEEPL = "deepl"
    GOOGLE = "google"
    OPENAI = "openai"

class FileUploadResponse(BaseModel):
    file_id: str
    filename: str
    page_count: int
    status: str

class TranslationRequest(BaseModel):
    file_id: str
    provider: TranslationProvider
    api_key: str
    source_lang: str = "en"
    target_lang: str = "zh"

class TranslationResponse(BaseModel):
    task_id: str
    status: str  # pending, processing, completed, failed
    progress: float  # 0-100
    message: Optional[str] = None

class PageContent(BaseModel):
    page_number: int
    text_blocks: List[TextBlock]
    images: List[ImageBlock]
    formulas: List[Formula]

class TextBlock(BaseModel):
    text: str
    bbox: Tuple[float, float, float, float]  # x0, y0, x1, y1
    font_size: float
    font_name: str
    is_bold: bool = False
    is_italic: bool = False

class ImageBlock(BaseModel):
    image_id: str
    bbox: Tuple[float, float, float, float]
    width: int
    height: int

class Formula(BaseModel):
    content: str  # LaTeX内容
    position: Tuple[int, int]  # 在文本中的位置
```

       │
